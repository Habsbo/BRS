	desc = br_mhc.1.desc

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			random_core_state = {
				limit = { is_controlled_by_ROOT_or_subject = yes }
				save_global_event_target_as = HMC_revolutionary_center
				set_state_flag = HMC_revolt_flag
				add_core_of = HMC
			}
			set_global_flag = br_global_hmc_revolt_flag
		}
	}

	option = {
		name = br_mhc.1.a #

		event_target:HMC_revolutionary_center = { add_core_of = HMC }
		
		hidden_effect = {
			random_list = {
				1 = { activate_mission = br_mhc_revolt_mission.missionary }
				1 = { activate_mission = br_mhc_revolt_mission.officer }
				1 = { activate_mission = br_mhc_revolt_mission.appealing }
			}
		}

		activate_mission = br_mhc_revolt_mission.uprising

		ai_chance = { base = 1 }
	}

}
