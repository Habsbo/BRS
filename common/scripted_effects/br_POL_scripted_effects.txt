POL_update_industrial_modernization_index = {
	if = {
		limit = { NOT = { has_variable = POL_investments_var } }
		set_variable = {
			var = POL_investments_var
			value = 1
		}
	}
	else = {
		add_to_variable = {
			var = POL_investments_var
			value = 1
		}
	}
}
