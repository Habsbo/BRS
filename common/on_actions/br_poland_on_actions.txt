				add_dynamic_modifier = {
					modifier = POL_spear_of_revenge_sanation_dynamic_modifier
				}

					modifier = POL_tight_grain_reserves_dynamic_modifier
				}
			}

				limit = { is_dynamic_country = no }
				if = {
					limit = {

				}
