# For minor countries where advisors are not defined
# Generate them from a generic template

# head of government
every_possible_country = {
	limit = {
		# avoid generic advisors in temporary countries
		is_dynamic_country = no
		NOT = {
			has_allowed_idea_with_traits = { idea = generic_head_of_gov characters = yes }
		}
	}
	generate_character = {
		token_base = generic_head_of_government # token will be TAG_token_base
		name = generic_head_of_government_name
		advisor = {
			slot = head_of_gov
			idea_token = generic_head_of_government
			traits = { generic_head_of_gov }
			can_be_fired = no
			cost = 0
			ai_will_do = {
				factor = 0
			}
		}
		portraits = { 
			army = { small = GFX_idea_generic_democratic_reformer }
		}
	}
}
